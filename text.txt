5. Create Ingest Route
Create:
src/api/routes/ingest.py
from fastapi import APIRouter, Header, HTTPException, Depends
from sqlalchemy.orm import Session
from src.schemas.ingest import IngestEvent
from src.database.connection import get_db
from src.services.ingest_service import process_event
router = APIRouter()
INGEST_API_KEY = "snx_demo_key"
@router.post("/api/public/ingest")
def ingest_event(
event: IngestEvent,
x_api_key: str = Header(...),
db: Session = Depends(get_db)
):
if x_api_key != INGEST_API_KEY:
raise HTTPException(status_code=401, detail="Invalid API key")
result = process_event(db, event)
return {
"success": True,
"event_id": result.id
}

7. Create Ingest Service
Create:
src/services/ingest_service.py
from src.database.models import LogEvent
from src.services.detection_service import run_detection
def process_event(db, event):
log = LogEvent(
timestamp=event.timestamp,
source_ip=event.source_ip,
destination_ip=event.destination_ip,
event_type=event.event_type,
severity=event.severity,
username=event.username,
status=event.status,
raw_data=event.dict()
)
db.add(log)
db.commit()
db.refresh(log)
run_detection(db, log)
return log
